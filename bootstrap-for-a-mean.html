1 Bootstrap for a Mean

We make skewed data, look at the mean, then see the bootstrap distribution.

set.seed(1)

# Simulate skewed data
x <- rlnorm(60, 3, 0.6)
hist(x, main = "Raw Data")
abline(v = mean(x), col = "red", lwd = 2)

# Plain mean
mean(x)
## [1] 24.19566
# Bootstrap: resample and compute mean many times
boot_means <- replicate(2000, mean(sample(x, replace = TRUE)))

# Show bootstrap distribution
hist(boot_means, main = "Bootstrap distribution of the mean",
     xlab = "Bootstrapped means")
abline(v = mean(x), col = "red", lwd = 2)

set.seed(123)

# Very skewed x
x <- c(rexp(40, rate = 0.1), 50, 60)      # skew + two leverage points
# Crazy noise that grows with x
y <- 5 + 0.5*x + rnorm(length(x), 0, sd = 5 + 0.2*x)

# Fit regression
fit <- lm(y ~ x)
summary(fit)

# Plot data + regression line
plot(x, y, main = "Skewed data with leverage points")

par(mfrow = c(1,2))
qqnorm(resid(fit), main = "QQ plot of residuals")
qqline(resid(fit))
plot(fitted(fit), resid(fit),
     xlab = "Fitted", ylab = "Residuals",
     main = "Residuals vs Fitted")
abline(h = 0, lty = 2)

par(mfrow = c(1,1))
confint(fit)["x", ]
##     2.5 %    97.5 % 
## 0.2970743 0.6866348
boot_slopes <- replicate(1000, coef(lm(y[sample(length(x), replace=TRUE)] ~ x[sample(length(x), replace=TRUE)]))[2])

quantile(boot_slopes, c(.025, .975))
##       2.5%      97.5% 
## -0.2344619  0.2949134
hist(boot_slopes, breaks = 30, main = "Bootstrap slopes", xlab = "Slope")
abline(v = coef(fit)[2], col = "red", lwd = 2)
